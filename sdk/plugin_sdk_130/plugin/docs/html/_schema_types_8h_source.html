117  inline bool16 IsBasicField(FieldType t)
118  {return ((t >= kBuiltInBase) && (t < (kBuiltInBase + kBuiltInArrayOffset)));}

120  {return ((t >= (kBuiltInBase + kBuiltInArrayOffset)) && (t < kStandardObjectsBase));}
