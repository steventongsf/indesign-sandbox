219  template <typename InputIt, typename OutputIt, typename CharType, typename EscapeType>
220  OutputIt Escape( InputIt srcBeg, InputIt srcEnd, OutputIt destBeg,
221  EscapeTable<CharType, EscapeType> const& mappingTable )
222  {
223  typedef typename EscapeTable<CharType, EscapeType>::EscapeMap MapType;
224  typedef typename EscapeTable<CharType, EscapeType>::EscapeMapKey MapKeyType;
225  typedef typename EscapeTable<CharType, EscapeType>::EscapeMapValue MapValueType;
226 
227  MapType const& searchMap = mappingTable.GetEscapeMap();
228  ASSERT( !searchMap.empty() );
229 
230  for ( ; srcBeg != srcEnd; ++srcBeg )
231  {
232  MapKeyType key(*srcBeg);
233  typename MapType::const_iterator i = searchMap.find( key );
234  if ( i != searchMap.end() )
235  {
236  destBeg = std::copy( i->second.begin(), i->second.end(), destBeg );
237  }
238  else
239  {
240  *destBeg++ = *srcBeg;
241  }
242  }
243  return destBeg;
244  }

260  template <typename InputIt, typename OutputIt, typename CharType, typename EscapeType>
261  OutputIt Unescape( InputIt srcBeg, InputIt srcEnd, OutputIt destBeg,
262  EscapeTable<CharType, EscapeType> const& mappingTable, bool16* pFoundInvalidEscape = nil )
263  {
264  typedef typename EscapeTable<CharType, EscapeType>::UnescapeMap MapType;
265  typedef typename EscapeTable<CharType, EscapeType>::UnescapeMapKey MapKeyType;
266  typedef typename EscapeTable<CharType, EscapeType>::UnescapeMapValue MapValueType;
267 
268  if ( pFoundInvalidEscape )
269  {
270  *pFoundInvalidEscape = kFalse;
271  }
272 
273  MapType const& searchMap = mappingTable.GetUnescapeMap();
274  ASSERT( !searchMap.empty() );
275 
276  MapKeyType const& escapeString = searchMap.begin()->first;
277 
278  typename MapKeyType::value_type escapeChar = escapeString[0];
279  size_t escapeLen = escapeString.size();
280 
281  while ( srcBeg != srcEnd )
282  {
283  // Is it an escape char?
284  if ( *srcBeg == escapeChar )
285  {
286  // Do we have enough chars left to be considered an escape sequence?
287  if ( escapeLen <= (size_t)std::distance(srcBeg, srcEnd) )
288  {
289  //MapKeyType key(srcBeg, srcBeg + escapeLen);
290  MapKeyType key;
291  std::copy(srcBeg, srcBeg + (int32)escapeLen, std::back_inserter(key) );
292  typename MapType::const_iterator i = searchMap.find( key );
293  if ( i != searchMap.end() )
294  {
295  // Replace with the found character
296  *destBeg++ = i->second;
297  std::advance(srcBeg, escapeLen);
298  // Continue with next char
299  continue;
300  }
301  }
302 
303  // We either found an invalid combination or a truncated escape sequence
304  // Signal that to the caller if it cares
305  if ( pFoundInvalidEscape )
306  {
307  *pFoundInvalidEscape = kTrue;
308  }
309  }
310 
311  // Copy the char to the output unchanged
312  *destBeg++ = *srcBeg++;
313  }
314 
315  return destBeg;
316  }

389  template<class PR> void StripBeginEndChars( WideString& text, PR filter, StringUtils::StripType stripType)
390  {
391  // remove leading characters
392  if (stripType == StringUtils::kStripLeadingAndTrailingChars || stripType == StringUtils::kStripLeadingChars)
393  {
394  WideString::const_iterator newBegin = text.begin();
395  for (; newBegin != text.end(); ++newBegin)
396  {
397  if (!filter(*newBegin))
398  break;
399  }
400  text.erase(text.begin(), newBegin);
401  }
402 
403  // remove trailing characters
404  if (stripType == StringUtils::kStripLeadingAndTrailingChars || stripType == StringUtils::kStripTrailingChars)
405  {
406  WideString::const_reverse_iterator rNewBegin = text.rbegin();
407  for (; rNewBegin != text.rend(); ++rNewBegin)
408  {
409  const UTF32TextChar c = *rNewBegin;
410  if (!filter(c))
411  break;
412  }
413 
414  text.erase(rNewBegin.base(), text.end());
415  }
416  }
