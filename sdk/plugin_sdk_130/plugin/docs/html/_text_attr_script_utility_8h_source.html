355 template<class T, class S> ErrorCode Accessor<T,S>::Access( IScriptRequestData* data, IScript* script, const ScriptID& propID )
356 {
357  ErrorCode err;
358  ScriptData scriptData;
359  if (data->IsPropertyGet())
360  {
361  InterfacePtr<const IAttrReport> attrReport;
362  err = fScriptUtils->GetAttributeReport(fAttrClass, attrReport);
363  if (err == kUnSetError)
364  {
365  scriptData.SetEnumeration(en_Nothing);
366  data->AppendReturnData( script, propID, scriptData ) ;
367  return kSuccess;
368  }
369 
370  if (err != kSuccess) return err;
371 
372  InterfacePtr<const T> attr(attrReport, UseDefaultIID());
373  if (!attr) return kFailure;
374 
375  return fMap.GetForGetProperty( script, data, propID, attr->Get() ) ;
376  }
377  else if (data->IsPropertyPut())
378  {
379  err = data->ExtractRequestData(propID, scriptData);
380  if (err != kSuccess) return err;
381 
382  if (Utils<IScriptUtils>()->IsNothing( scriptData )) // removing an attribute
383  {
384  err = fScriptUtils->RemoveAttribute( fAttrClass ) ;
385  }
386  else // adding/modifying an attribute
387  {
388  int32 val(0);
389  err = fMap.GetForSetProperty( script, data, propID, val ) ;
390  if (err != kSuccess) return err;
391 
392  InterfacePtr<T> attrR(::CreateObject2<T>(fAttrClass));
393  if (!attrR) return kFailure;
394 
395  attrR->Set(static_cast<S>(val));
396  err = fScriptUtils->AddAttribute( attrR ) ;
397  }
398 
399  return err ;
400  }
401 
402  return kFailure ;
403 }
