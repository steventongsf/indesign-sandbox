355  Font_Info(const CTFontDict* ctFontDict, int32 numDesignVectors = 0, Fixed* designVectors = nil, uint32 flags = 0)
356  : fctFontDict(ctFontDict),

362  fDesignVectors[i] = 0;

365  ASSERT(numDesignVectors <= 4);
366  if (fnumDesignVectors > 4)
367  fnumDesignVectors = 4;

369  fDesignVectors[i] = designVectors[i];
370  }
371  }

397  Font_Info &operator= (const Font_Info& other)

402  fnumDesignVectors = other.fnumDesignVectors;

417  bool16 operator==(const Font_Info& fi) const
418  {
419  if (fctFontDict != fi.GetCTFontDict())
420  return kFalse;
421  if (fnumDesignVectors != fi.GetNumDesignVectors())
422  return kFalse;
423  Fixed fiDV[4];

425  for (int32 i = 0; i < fnumDesignVectors; i++)
