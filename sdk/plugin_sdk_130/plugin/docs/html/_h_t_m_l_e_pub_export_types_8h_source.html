193  IDStyleType GetStyleType() const { return fType; }
194  PMString GetPath () const { return fPath; }
195 
197  bool16 operator==(const IDStyle &other) const
198  {
199  return (fType == other.fType) && (fPath == other.fPath);
200  }
201 
203  bool16 operator!=(const IDStyle &other) const
204  {
205  return (fType!=other.fType) || (fPath!=other.fPath);
206  }
207 
209  bool16 operator <(const IDStyle & other) const
210  {
211  if (fType == other.fType)
212  return fPath < other.fPath;
213  else
214  return fType < other.fType;
215  }
216 
217  operator void* () const // nonzero if good
