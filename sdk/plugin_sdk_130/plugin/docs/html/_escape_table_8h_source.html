65 //
66 // Wrapper over a C-array of EscapePairs
67 //
68 // WARNING: this class does NOT copy the data passed in the constructor.
69 // It is rather a wrapper over a C-array of EscapePairs.
70 // The array that is passed in the constructor has to be valid for the lifetime of this object.
71 //
72 // All escape sequences must start with the same escape char AND they must either be the same length
73 // or they should have the same ending escape char.
74 //
75 // T is the character type for the character to be escaped
76 // U is the character type of the escape sequence
77 //
78 // The escape tables can come in two flavors:
79 // - all escape sequences have the same start escape marker and fixed length (the end escape marker doesn't exist)
80 // - all escape sequences have the same pair of start and end markers and they can be of different lengths.
81 //
82 
83 // EscapeTable is usually a shared const variable.
84 // However, initialization of the members can lead to undesirable
85 // race conditions unless synchronization is implemented.
86  void* AcquireEscapeTableLock();
87  void ReleaseEscapeTableLock(void* lock);
88 template<typename T, typename U>
89 class EscapeTable
90 {
91  // This class is not copyable
92  EscapeTable(const EscapeTable &);
93  EscapeTable& operator=(const EscapeTable &);
94 
95 
96 public:
97  typedef size_t size_type;
98  typedef EscapePair<T, U> value_type;
99  typedef EscapePair<T, U>& reference;
100  typedef EscapePair<T, U> const& const_reference;
101 // typedef std::map<T, std::basic_string<U> > EscapeMap;
102 // typedef std::map<std::basic_string<U>, T> UnescapeMap;
103  // Determine which char type is larger
104  typedef typename SelectType<(sizeof(T) > sizeof(U)), T, U>::Result BigCharType;
105 
106  // Select the bigger char type as lookup key and the smaller one as the value
107  // This will make it easier with the casts
108  // (always better to go from char->wide char take advantage of type promotion)
109  typedef BigCharType EscapeMapKey;
110  typedef typename EscapePair<T, U>::char_range_ref EscapeMapValue;
111 
112  typedef typename std::basic_string<BigCharType> UnescapeMapKey;
113  typedef T UnescapeMapValue;
114 
115  typedef typename std::map<EscapeMapKey, EscapeMapValue> EscapeMap;
116  typedef typename std::map<UnescapeMapKey, UnescapeMapValue> UnescapeMap;
117 
121  EscapeTable( EscapePair<T, U> escapes[], size_type count )
122  : array_(escapes), size_(count), endEscapeMarker_(U())
123  {
124  // Alias for easier reading
125  EscapeMapValue const& escapeSequence( array_[0].escapeSeq );
126  ASSERT( !escapeSequence.empty() );
127 
128  // Save the start escape marker (all start escape markers should be the same)
129  startEscapeMarker_ = escapeSequence.front();
130 
131  ASSERT( IsValid() );
132  }
133 
138  EscapeTable( EscapePair<T, U> escapes[], size_type count, U startEscapeMarker, U endEscapeMarker )
139  : array_(escapes), size_(count), startEscapeMarker_(startEscapeMarker), endEscapeMarker_(endEscapeMarker)
140  {
141  // The end marker can not be the nil character
142  ASSERT( endEscapeMarker_ != U() );
143  ASSERT( IsValid() );
144  }
145 
146  bool16 IsValidEscapeSequence( const EscapeMapValue& escapeSequence ) const
147  {
148  ASSERT( !escapeSequence.empty() );
149  // All escape sequences must start with the same escape char
150  // AND they either must be the same length or they should have the same ending escape char
151 
152  if ( endEscapeMarker_ != U() )
153  {
154  return (escapeSequence.front() == startEscapeMarker_ && escapeSequence.back() == endEscapeMarker_);
155  }
156 
157  // Must have the same length
158  return ( escapeSequence.front() == startEscapeMarker_ && escapeSequence.size() == array_[0].escapeSeq.size() );
159  }
160 
161  // Checks if this table is valid
162  bool16 IsValid() const
163  {
164  ASSERT( size() );
165  bool16 result( true );
166 
167  size_type i( 0 );
168  for (; i < size() && result; ++i )
169  {
170  // all escape sequences must start with the same escape char
171  // AND they either must be the same length or they should have the same ending escape char
172  result = IsValidEscapeSequence(array_[i].escapeSeq);
173  }
174  return result;
175  }
176 
177  // read-only access for an EscapePair at the specified index
178  const_reference operator[]( size_type index ) const
179  {
180  return array_[index];
181  }
182 
183  // number of elements in the array
184  size_type size() const
185  {
186  return size_;
187  }

189  EscapeMap const& GetEscapeMap() const
190  {
191  if (!escapeMap_.get() )
192  {
193  void* lock = AcquireEscapeTableLock();
194  if ( !escapeMap_.get() )
195  escapeMap_.reset( CreateEscapeMap());
196  ReleaseEscapeTableLock(lock);
197  }
198  return *escapeMap_;
199  }
200 
201  UnescapeMap const& GetUnescapeMap() const
202  {
203  if ( !unescapeMap_.get() )
204  {
205  void* lock = AcquireEscapeTableLock();
206  if ( !unescapeMap_.get() )
207  {
208  unescapeMap_.reset( CreateUnescapeMap());
209  }
210  ReleaseEscapeTableLock(lock);
211  }
212  return *unescapeMap_;
213  }

216 private:
217 
218  EscapeMap* CreateEscapeMap( ) const
219  {
220  ASSERT( IsValid() );
221 
222  bool16 result(true);
223  EscapeMap* escapeMap = new EscapeMap;
224  size_type i(0);
225  for (; i < size() && result; ++i )
226  {
227  result = escapeMap->insert(
228  std::make_pair(array_[i].ch, array_[i].escapeSeq) ).second;
229  ASSERT_MSG( result, "CreateEscapeMap failed to insert escape seq! Check for duplicates!" );
230  }
231 
232  if (!result)
233  {
234  delete escapeMap;
235  escapeMap = nil;
236  }
237  return escapeMap;
238  }
239 
240  UnescapeMap* CreateUnescapeMap( ) const
241  {
242  ASSERT( IsValid() );
243  UnescapeMap* unescapeMap = new UnescapeMap;
244 
245  bool16 result(true);
246  size_type i(0);
247  for (; i < size() && result; ++i )
248  {
249  UnescapeMapKey key(array_[i].escapeSeq.begin(), array_[i].escapeSeq.end());
250 
251  result = unescapeMap->insert(
252  std::make_pair(key, array_[i].ch) ).second;
253  ASSERT_MSG( result, "CreateUnescapeMap failed to insert unescape seq! Check for duplicates!" );
254  }
255 
256  if ( !result )
257  {
258  delete unescapeMap;
259  unescapeMap = nil;
260  }
261  return unescapeMap;
262  }

272 };
