44 // The point of this allocator is to be able to use an STL object while completely
45 // bypassing our allocation mechanisms and going directly to the runtime library.
46 // This is done by using malloc/free in the underlying RTL.
47 
48 template <class T>
49 class BypassAllocator {
50 public:
51  // type definitions
52  typedef T value_type;
53  typedef T* pointer;
54  typedef const T* const_pointer;
55  typedef T& reference;
56  typedef const T& const_reference;
57  typedef size_t size_type;
58  typedef ptrdiff_t difference_type;
59 
60  // rebind allocator to type U
61  template <class U >
62  struct rebind {
63  typedef BypassAllocator< U > other;
64  };
65 
66  // return address of values
67  pointer address (reference value) const {
68  return &value;
69  }
70  const_pointer address (const_reference value) const {
71  return &value;
72  }
73 
74  /* constructors and destructor
75  * - nothing to do because the allocator has no state
76  */
77  BypassAllocator() noexcept {
78  }
79  BypassAllocator(const BypassAllocator& src) noexcept {
80  }
81  template <class U >
82  BypassAllocator (const BypassAllocator< U > &src) noexcept {
83  }
84  ~BypassAllocator() noexcept {
85  }
86 
87  // return maximum number of elements that can be allocated
88  size_type max_size () const noexcept {
89  return (size_type) (int64) -1 / sizeof(T);
90  }
91 
92  // allocate but don't initialize num elements of type T
93  pointer allocate (size_type num, const void* = 0) {
94 #if defined(MACINTOSH) && defined(DEBUG)
95  pointer ret = (pointer) K2Memory::RTLCompatibleBypassMalloc (num * sizeof(T));
96 #else
97  pointer ret = (pointer) malloc (num * sizeof(T));
98 #endif
99  if (!ret) throw std::bad_alloc();
100  return ret;
101  }
102 
103  // initialize elements of allocated storage p with value value
104  void construct (pointer p, const T& value) {
105  // initialize memory with placement new
106  new((void*)p)T(value);
107  }
108 
109  // destroy elements of initialized storage p
110  void destroy (pointer p) {
111  // destroy objects by calling their destructor
112  p->~T();
113  }
114 
115  // deallocate storage p of deleted elements
116  void deallocate (pointer p, size_type num) {
117 #if defined(MACINTOSH) && defined(DEBUG)
118  K2Memory::RTLCompatibleBypassFree (p);
119 #else
120  free (p);
121 #endif
122  }
123 
124 };

127 template <class T1, class T2>
128 bool operator== (const BypassAllocator<T1>&, const BypassAllocator<T2>&) noexcept {
129  return true;
130 }
131 template <class T1, class T2>
132 bool operator!= (const BypassAllocator<T1>&, const BypassAllocator<T2>&) noexcept {
133  return false;
134 }

138 // This is similar, but keeps a pool of recently used objects to recycle.
139 
140 // This is cheesy and NOT a good example of how to do this in a general case for
141 // a number of reasons, but this is sufficient for our memory tracking debug code.
142 
143 template <class T>
144 class BypassRecycleAllocator {
145 public:
146  // type definitions
147  typedef T value_type;
148  typedef T* pointer;
149  typedef const T* const_pointer;
150  typedef T& reference;
151  typedef const T& const_reference;
152  typedef size_t size_type;
153  typedef ptrdiff_t difference_type;
154 
155  // rebind allocator to type U
156  template <class U >
157  struct rebind {
158  typedef BypassRecycleAllocator< U > other;
159  };
160 
161  // return address of values
162  pointer address (reference value) const {
163  return &value;
164  }
165  const_pointer address (const_reference value) const {
166  return &value;
167  }
168 
169  /* constructors and destructor
170  */
171  BypassRecycleAllocator() noexcept : fSP (0) {
172  }
173  BypassRecycleAllocator(const BypassRecycleAllocator& src) noexcept : fSP (0) {
174  }
175  template <class U >
176  BypassRecycleAllocator (const BypassRecycleAllocator< U > &src) noexcept : fSP (0) {
177  }
178  ~BypassRecycleAllocator() noexcept {
179  while (fSP)
180 #if defined(MACINTOSH) && defined(DEBUG)
181  K2Memory::RTLCompatibleBypassFree (fStack [--fSP]);
182 #else
183  free (fStack [--fSP]);
184 #endif
185  }
186 
187  // return maximum number of elements that can be allocated
188  size_type max_size () const noexcept {
189  return (size_type) (int64) -1 / sizeof(T);
190  }
191 
192  // allocate but don't initialize num elements of type T
193  pointer allocate (size_type num, const void* = 0) {
194  if (num == 1 && fSP) return fStack [--fSP];
195 #if defined(MACINTOSH) && defined(DEBUG)
196  pointer ret = (pointer) K2Memory::RTLCompatibleBypassMalloc (num * sizeof(T));
197 #else
198  pointer ret = (pointer) malloc (num * sizeof(T));
199 #endif
200  if (!ret) throw std::bad_alloc();
201  return ret;
202  }
203 
204  // initialize elements of allocated storage p with value value
205  void construct (pointer p, const T& value) {
206  // initialize memory with placement new
207  new((void*)p)T(value);
208  }
209 
210  // destroy elements of initialized storage p
211  void destroy (pointer p) {
212  // destroy objects by calling their destructor
213  p->~T();
214  }
215 
216  // deallocate storage p of deleted elements
217  void deallocate (pointer p, size_type num) {
218  if (num == 1 && fSP < kMaxSize)
219  fStack [fSP++] = p;
220  else
221 #if defined(MACINTOSH) && defined(DEBUG)
222  K2Memory::RTLCompatibleBypassFree (p);
223 #else
224  free (p);
225 #endif
226  }
227 
228 private:
229 
230  enum { kMaxSize = 1000 }; // tested: 90% recycle rate for build acceptance, only 10% of requests actually hit malloc/free
231  pointer fStack [kMaxSize];
232  int32 fSP;
233 };
