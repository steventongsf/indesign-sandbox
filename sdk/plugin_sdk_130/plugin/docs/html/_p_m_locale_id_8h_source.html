215 // if either is wild or they match
216 inline bool16 PMLocaleId::IsProductFS(const int16 productFS) const
217 {
218  uint16 product = static_cast<uint16>(fFeatureSetId) & 0xFF00;
219 #if 1
220  return ((product == kAllProductsFS) || (static_cast<uint16>(productFS) == kAllProductsFS) || (product == static_cast<uint16>(productFS)));
221 #else
222  //11/24/03 pboctor/mburbidge - The last part of this expression was changed to allow a ScriptInfo resource to be something like this:
223  // kInDesignRomanFS | kInDesignServerRomanFS
224  // In order for this to work: 1. The product feature set values must occupy unique bits 2. We can't have a product feature set value of 0
225  //5/13/04 jwbrown - ScriptInfo resources no longer require/permit OR'd feature sets, so I've reverted this code
226  return ((product == (int16)kAllProductsFS) || (productFS == (int16)kAllProductsFS) || ((productFS & product) != 0));
227 #endif
228 }

231 inline bool16 PMLocaleId::IsLanguageFS(const int16 languageFS) const
232 {
233  int16 language = fFeatureSetId & 0x00FF;
234  return ((language == (int16)kAllLanguagesFS) || (languageFS == (int16)kAllLanguagesFS) || (language == languageFS));
